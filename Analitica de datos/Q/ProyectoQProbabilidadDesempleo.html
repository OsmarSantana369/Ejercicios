# Por cada columna, mostrar las medidas de tendencia central y dispersin, asi como una representacin grfica, omitiendo los valores nulos
for column in data.columns:
	print(f"Anlisis de la variable: {column}")
	print(f"Media: {data[column].mean()}")
	print(f"Mediana: {data[column].median()}")
	print(f"Moda: {data[column].mode()[0]}")
	print(f"Desviacin estndar: {data[column].std()}")
	print(f"Varianza: {data[column].var()}")
	print(f"Mnimo: {data[column].min()}")
	print(f"Mximo: {data[column].max()}")
	print(f"Rango: {data[column].max() - data[column].min()}")
	print("\n")
	
	# Cada fila representa un ao (las etiquetas deben empezar desde 2000), por lo que haremos un grfico de barras (usando sns) para cada variable
	plt.figure(figsize=(10, 6))
	sns.barplot(x=range(2000, 2000 + len(data.index)), y=data[column])
	plt.title(f"{column} a lo largo de los aos")
	plt.xlabel("Ao")
	plt.ylabel("")
	plt.xticks(rotation=45)
	plt.show()
